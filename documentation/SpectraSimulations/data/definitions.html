123    def filterLevel(self, low_level: str, high_level: str, auger_level: str = '', *, strict: str) -> bool:
124        if auger_level == '':
125            if strict != 'h' and strict != 'l' and strict != 'hl' and strict != 'na' and strict != 'hia':
126                raise RuntimeError("Error: Wrong parameters passed to filterLevel function of Line (Radiative).")
127        else:
128            if strict != 'h' and strict != 'l' and strict != 'hl' and strict != 'na' and strict != 'ha' and strict != 'la' and strict != 'hla' and strict != 'hia':
129                raise RuntimeError("Error: Wrong parameters passed to filterLevel function of Line (Auger).")
130        
131        if self.energy == 0:
132            return False

134        if strict == 'h':
135            if auger_level == '':
136                return (self.Shelli in low_level or low_level in self.Shelli) and self.Shellf == high_level

139        elif strict == 'hia':

144        elif strict == 'l':

149        elif strict == 'hl':

156                return (self.Shelli in low_level or low_level in self.Shelli) and (self.Shellf in high_level or high_level in self.Shellf)

169    def filterInitialState(self, *args) -> bool:
170        if len(args) == 1:
171            if isinstance(args[0], list) or isinstance(args[0], tuple):
172                if len(args[0]) == 3:
173                    return self.Shelli == args[0][0] and self.jji == args[0][1] and self.eigvi == args[0][2]
174            elif isinstance(args[0], Line):
175                return self.Shelli == args[0].Shelli and self.jji == args[0].jji and self.eigvi == args[0].eigvi
176        elif len(args) == 3:
177            return self.Shelli == args[0] and self.jji == args[1] and self.eigvi == args[2]
178        
179        print("Warning: bad arguments at initial state match: " + self.Shelli + ", " + str(self.jji) + ", " + str(self.eigvi))
180        print("With arguments: " + str(args))
181        return False

215    def match(self, other):
216        if hasattr(self, 'Shellf') and hasattr(other, 'Shellf'):
217            if self.Shelli == other.Shelli and self.jji == other.jji and self.eigvi == other.eigvi \
218               and self.Shellf == other.Shellf and self.jjf == other.jjf and self.eigvf == other.eigvf :
219                return True
220        elif hasattr(self, 'Shellf') and not hasattr(other, 'Shellf') or not hasattr(self, 'Shellf') and hasattr(other, 'Shellf'):
221            return False
222        else:
223            if self.Shelli == other.Shelli and self.jji == other.jji and self.eigvi == other.eigvi:
224                return True

226    def effectiveIntensity(self, beam: float, FWHM: float, crossSection,
227                           include_cascades: bool, boost_type: str, key: str = '',
228                           shake_amps: dict = {}, shakeoff_lines: List[List[str]] | None = [],
229                           shakeup_lines: List[List[str]] | None = [],
230                           shakeup_splines = {}, shake_missing: Dict[str, float] = {},
231                           alpha: float = 1.0, exc_index: int = -1) -> float:
232        from simulation.shake import calculateTotalShake, get_shakeoff, get_shakeup
233        from simulation.shake import get_shakeoff_exc, get_shakeup_exc
234        from simulation.mults import get_overlap, get_overlap_exc
235        
236        if boost_type == 'diagram':
237            boostDict = generalVars.radBoostMatrixDict
238        elif boost_type == 'auger':
239            boostDict = generalVars.augBoostMatrixDict
240        elif boost_type == 'satellite':
241            boostDict = generalVars.satBoostMatrixDict
242        elif boost_type == 'shakeup':
243            boostDict = generalVars.satBoostMatrixDict #temp solution
244        else:
245            print("Error on selected boost type for effective intensity calculation: " + boost_type)
246            raise RuntimeError("Available boost types as diagram, auger and satellite")
247        
248        crossKey = self.Shelli if key == '' else key
249        
250        beamEnergy = beam if beam > 0.0 else generalVars.defaultBeam
251        
252        absIntensity = self.intensity
253        boostMult = boostDict[self.key()] if include_cascades and self.key() in boostDict else 0.0
254        if exc_index == -1:
255            overlapMult = get_overlap(self, beamEnergy, FWHM) if beam > 0.0 else 1.0
256        else:
257            overlapMult = get_overlap_exc(self, beamEnergy, FWHM, exc_index) if beam > 0.0 else 1.0
258        crossMult = crossSection[crossKey](generalVars.formationEnergies[boost_type][self.keyI()], generalVars.defaultBeam) if type(crossSection) != type(1.0) and beam <= 0.0 else crossSection
259        mixMult = self.mix if hasattr(self, 'mix') else 1.0
260        
261        if generalVars.shakeoff[0] != [''] and generalVars.shakeup[0] != ['']:
262            diagramMult = (1 - calculateTotalShake(self.jji, shake_amps, shakeoff_lines, shakeup_lines)) if boost_type == 'diagram' or boost_type == 'auger' else 1.0
263        else:
264            diagramMult = 1.0
265        
266        if exc_index == -1:
267            shakeoffMult = self.diagramOverlap * get_shakeoff(key, shakeoff_lines) if boost_type == 'satellite' else 1.0
268            shakeupMult = self.diagramOverlap * get_shakeup(key, self.Shelli[4:], self.jji, shakeup_splines, shake_missing) if boost_type == 'shakeup' else 1.0
269        else:
270            # shakeoffMult = self.diagramOverlap * get_shakeoff_exc(key, exc_index, shakeoff_lines) if boost_type == 'satellite' else 1.0
271            shakeoffMult = 1.0
272            # shakeupMult = self.diagramOverlap * get_shakeup_exc(key, self.Shelli[4:], self.jji, exc_index, shakeup_splines, shake_missing) if boost_type == 'shakeup' else 1.0
273            shakeupMult = 1.0
274        if len(shake_amps) > 0:
275            shakeoffMod = shake_amps['shake_amp_' + key] if "shake_amp_" + key in shake_amps else 1.0 if boost_type == 'satellite' else 1.0
276            shakeupMod = shake_amps['shakeup_amp_' + key] if "shakeup_amp_" + key in shake_amps else 1.0 if boost_type == 'shakeup' else 1.0
277        else:
278            shakeoffMod = 1.0
279            shakeupMod = 1.0
280        
281        # if boost_type == 'satellite':
282        #     generalVars.control_shakeoff[key] = get_shakeoff(key)
283        # if boost_type == 'shakeup':
284        #     generalVars.control_shakeup[key + "_" + str(self.jji) + "_" + self.Shelli[4:-1]] = get_shakeup(key, self.Shelli[4:], self.jji)
285        
286        # if boost_type == 'satellite':# and shakeupMult > 0.0:# and 'M2' in self.Shelli:
287        #     # with open("shakeup_debug.txt", "a") as f:
288        #     print(self.key() + ": " + \
289        #     str(absIntensity) + ', ' + \
290        #     str(crossMult) + ', ' + \
291        #     str(diagramMult) + ', ' + \
292        #     str(mixMult) + ', ' + \
293        #     str(overlapMult) + ', ' + \
294        #     str(shakeoffMod) + ', shakeMult -> ' + \
295        #     str(shakeoffMult) + ', ' + \
296        #     str(shakeupMod) + ', ' + \
297        #     str(shakeupMult) + ', ' + \
298        #     str(boostMult))
299        
300        # Effective intensity = direct decay + cascade decay (boost mult)
301        return absIntensity * \
302            (crossMult * diagramMult * mixMult * overlapMult * \
303            shakeoffMod * shakeoffMult * shakeupMod * shakeupMult + \
304            boostMult) * \
305            alpha
