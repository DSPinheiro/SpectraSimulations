 38def stem_ploter(a: Axes, x: List[float], y: List[float], JJ: List[int], transition: str, spec_type: str, ind: int = 0, key: str = '', x_up: List[float] = [], y_up: List[float] = [], JJ_up: List[int] = []):
 39    """
 40    Stick plotter function. Plots a stick for the transition
 41        
 42        Args:
 43            a: matplotlib axes object where to plot the stems
 44            x: list of the transition energies
 45            y: list of the transition intensities
 46            JJ: list of the transition 2js
 47            transition: the selected transition to plot
 48            spec_type: simulation type selected in the interface (diagram, satellite, auger, diagram_cs, satellite_cs, auger_cs)
 49            ind: shake level index to use when plotting a satellite transition
 50            key: shake level label to use when plotting a satellite transition
 51            x_up: list of the shake-up transition energies
 52            y_up: list of the shake-up transition intensities
 53            JJ_up: list of the shake-up transition 2js
 54        
 55        Returns:
 56            Nothing, the transition is plotted and the interface is updated
 57    """
 58    
 59    JJs = list(set(JJ))
 60    JJs_up = list(set(JJ_up))
 61    
 62    ys: List[List[float]] = [[] for _ in JJs]
 63    ys_up: List[List[float]] = [[] for _ in JJs_up]
 64    xs: List[List[float]] = [[] for _ in JJs]
 65    xs_up: List[List[float]] = [[] for _ in JJs_up]
 66    
 67    if guiVars.JJ_colors.get(): # type: ignore
 68        for h, j in enumerate(JJ):
 69            if j not in generalVars.colors_2J:
 70                generalVars.colors_2J[j] = select_color()
 71
 72            i = JJs.index(j)
 73            ys[i].append(y[h])
 74            xs[i].append(x[h])
 75        
 76        for h, j in enumerate(JJ_up):
 77            if j not in generalVars.colors_2J:
 78                generalVars.colors_2J[j] = select_color()
 79
 80            i = JJs_up.index(j)
 81            ys_up[i].append(y_up[h])
 82            xs_up[i].append(x_up[h])
 83    
 84    # Add extra values before and after to make the y start and terminate on 0
 85    max_value = max(x) if len(x) > 0 else 0.0
 86    min_value = min(x) if len(x) > 0 else 0.0
 87    delta = (min_value + max_value) / 2.0
 88    
 89    x_tmp = x.copy()
 90    
 91    if guiVars.x_min.get() == "Auto": # type: ignore
 92        x_tmp.insert(0, min_value - delta * 0.2)
 93        y.insert(0, 0)

 95        a.set_xlim(xmin=float(guiVars.x_min.get())) # type: ignore

109    if spec_type == 'Diagram' or spec_type == 'Auger':
110        with open("debug_sticks.txt", "a") as debug:
111            debug.write("\nDiagram Sticks:\n\n")
112            debug.write(tr_readable + "\n")
113        if not guiVars.JJ_colors.get(): # type: ignore
114            with open("debug_sticks.txt", "a") as debug:
115                for x1, y1 in zip(x_tmp, y):
116                    debug.write(str(x1) + ", " + str(y1) + "\n")
117            a.stem(x_tmp, y, markerfmt=' ', linefmt=select_color(), label=tr_latex, use_line_collection=True)
118            a.legend(loc='best', numpoints=1)
119        else:
120            for i in range(len(xs)):
121                a.stem(xs[i], ys[i], markerfmt=' ', linefmt=str(generalVars.colors_2J[JJ[i]]), label=tr_latex + "_" + str(JJ[i]), use_line_collection=True)
122                a.legend(loc='best', numpoints=1)
123    elif spec_type == 'Satellites':
124        if not guiVars.JJ_colors.get(): # type: ignore
125            if len(x_tmp) > 0 and len(y) > 0:
126                with open("debug_sticks.txt", "a") as debug:
127                    debug.write("\nShake-off Sticks:\n\n")
128                    debug.write(tr_readable + ' - ' + generalVars.labeldict[key] + "\n")
129                    for x1, y1 in zip(x_tmp, y):
130                        debug.write(str(x1) + ", " + str(y1) + "\n")
131                a.stem(x_tmp, y, markerfmt=' ', linefmt=select_color(), label=tr_latex + ' - ' + generalVars.labeldict[key], use_line_collection=True)  # Plot a stemplot
132                a.legend(loc='best', numpoints=1)
133            
134            if len(x_up) > 0 and len(y_up) > 0:
135                with open("debug_sticks.txt", "a") as debug:
136                    debug.write("\nShake-up Sticks:\n\n")
137                    debug.write(tr_readable + ' - ' + generalVars.labeldict[key] + ' - shakeup\n')
138                    for x1, y1 in zip(x_up, y_up):
139                        debug.write(str(x1) + ", " + str(y1) + "\n")
140                a.stem(x_up, y_up, markerfmt=' ', linefmt=select_color(), label=tr_latex + ' - ' + generalVars.labeldict[key] + ' - shakeup', use_line_collection=True)  # Plot a stemplot
141                a.legend(loc='best', numpoints=1)
142        else:
143            if len(xs) > 0 and len(ys) > 0 and len(JJ) > 0:
144                for i in range(len(xs)):
145                    a.stem(xs[i], ys[i], markerfmt=' ', linefmt=str(generalVars.colors_2J[JJ[i]]), label=tr_latex + ' - ' + generalVars.labeldict[key] + '_' + str(JJ[i]), use_line_collection=True)  # Plot a stemplot
146                    a.legend(loc='best', numpoints=1)

148            if len(x_up) > 0 and len(y_up) > 0 and len(JJ_up) > 0:
149                for i in range(len(xs_up)):
150                    a.stem(xs_up[i], ys_up[i], markerfmt=' ', linefmt=str(generalVars.colors_2J[JJ_up[i]]), label=tr_latex + ' - ' + generalVars.labeldict[key] + ' - shakeup_' + str(JJ_up[i]), use_line_collection=True)  # Plot a stemplot
151                    a.legend(loc='best', numpoints=1)
152    elif spec_type == 'Diagram_CS' or spec_type == 'Auger_CS':
153        if not guiVars.JJ_colors.get(): # type: ignore
154            a.stem(x_tmp, y, markerfmt=' ', linefmt=select_color(), label=tr_latex, use_line_collection=True)
155            a.legend(loc='best', numpoints=1)

157            for i in range(len(xs)):
158                a.stem(xs[i], ys[i], markerfmt=' ', linefmt=str(generalVars.colors_2J[JJ[i]]), label=tr_latex + "_" + str(JJ[i]), use_line_collection=True)

166            a.legend(loc='best', numpoints=1)

170                a.legend(loc='best', numpoints=1)

173                a.legend(loc='best', numpoints=1)

175    # --------------------------------------------------------------------------------------------------------------------------
176    # Automatic legend formating
177    a.legend(title=generalVars.element_name, title_fontsize='large')
178    a.legend(title=generalVars.element_name)
179    # Number of total labels to place in the legend
180    number_of_labels = len(a.legend().get_texts())

191        labels_per_columns = number_of_labels / legend_columns
192    
193    # Place the legend with the final number of columns
194    a.legend(ncol=legend_columns)
195    
196    return a
